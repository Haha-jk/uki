

program  
    statement*

statement  
    container-statement
    expression-statement 

container-statement
    container-declaration  
    container-retrieval

container-declaration 
    identifier parameters? ":" expression

container-retrieval
    identifier arguments?

parameters
    identifier ("," identifier)* 

arguments  
    expression ("," expression)*  


identifier
    [a-z] ([a-z | 0-9 | "-"] [a-z | 0-9])*




variable
    lower |
    lower rest

rest
    "-" (segment | digit "-" segment)
    
segment
    lower |
    lower rest
